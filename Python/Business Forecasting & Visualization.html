# import libraries
import pandas as pd
import matplotlib.pyplot as plt

# load dataset
df = pd.read_csv("Clean_dataset.csv")

# define readmission order
readmit_order = ["NO", ">30", "<30"]
df["readmitted"] = pd.Categorical(df["readmitted"], categories=readmit_order, ordered=True)

# create early readmission flag
df["early_readmit_flag"] = df["readmitted"].apply(lambda x: 1 if x == "<30" else 0)

# -------------------------------
# 1. age group vs early readmission rate
# -------------------------------
age_rate = (
    df.groupby("age", observed=False)["early_readmit_flag"]
    .mean()
    .reset_index()
)

age_rate["early_readmission_rate_pct"] = age_rate["early_readmit_flag"] * 100
age_rate = age_rate[["age", "early_readmission_rate_pct"]]

print("Early Readmission Rate by Age Group")
print(age_rate)

plt.figure(figsize=(10, 5))
plt.bar(age_rate["age"].astype(str), age_rate["early_readmission_rate_pct"])
plt.title("Early Readmission Rate by Age Group")
plt.xlabel("Age Group")
plt.ylabel("Early Readmission Rate (%)")
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

        diagnoses.loc[diagnoses["readmitted"] == ">30", "number_diagnoses"].values[0]
    ],
    "<30": [
        time_hospital.loc[time_hospital["readmitted"] == "<30", "time_in_hospital"].values[0],
